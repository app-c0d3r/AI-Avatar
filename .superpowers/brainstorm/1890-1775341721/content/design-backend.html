Change 2 — Buffer wrapper function (new)

async def emo_buffered_stream(raw_gen):
  buffer = ""
  tag_done = False
  async for chunk in raw_gen:
    # unwrap existing data: {...} chunks
    text = extract_content(chunk)
    if not tag_done:
      buffer += text
      if "]" in buffer:
        # tag complete → parse & emit
        yield from emit_emo_and_flush(buffer)
        tag_done = True
      elif len(buffer) > 60:
        # fallback: LLM forgot tag
        yield emo_event("neutral", "medium")
        yield msg_event(buffer)
        tag_done = True
    else:
      yield msg_event(text)
  yield "data: [DONE]\n\n"

SSE wire format — before vs. after

BEFORE (current)
data: {"content": "Hello"}
data: {"content": " there"}
data: [DONE]
No emotion info. Frontend guesses from keywords.
AFTER (new)
event: emotion
data: {"emotion":"happy","intensity":"medium"}

event: message
data: {"content": "Hello"}

event: message
data: {"content": " there"}

data: [DONE]
