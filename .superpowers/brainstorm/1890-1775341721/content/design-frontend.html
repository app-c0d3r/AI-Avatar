useFrame — three-layer execution order

// ── LAYER 1: Pose (arms + spine + head tilt) ─────────────────
const targets = getPoseTargets(poseRef, intensityRef, t)
b.lUAz = lerp(b.lUAz, targets.lUAz, 0.08)
// ... lLAz, rUAz, rLAz same as before ...
b.spineX = lerp(b.spineX, targets.spineX, 0.06) // slightly slower for organic feel
b.headZ = lerp(b.headZ, targets.headZ, 0.05)

// ── LAYER 2: Talk gestures (additive during TTS) ─────────────
if (isTalkingRef.current && volume > 10) {
  const v = Math.min(volume / 128, 1)
  b.rLAz += Math.sin(t * 4) * 0.08 * v // wrist flick
  b.lUAz += Math.sin(t * 2.5 + 1) * 0.04 * v // left arm breathes
}

// ── LAYER 3: Existing (mouse, blink, mouth) ── unchanged ─────

// ── Apply all bones ──────────────────────────────────────────
leftArm.rotation.z = b.lUAz; rightArm.rotation.z = b.rUAz
leftLower.rotation.z = b.lLAz; rightLower.rotation.z = b.rLAz
spine?.rotation.x = b.spineX
head.rotation.z = b.headZ // additive on top of mouse Y/X
